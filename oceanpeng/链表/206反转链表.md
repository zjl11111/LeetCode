/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        ListNode *pre = NULL,*now = head,*tmp = NULL;
        if(now == NULL) return now;
        while(now -> next != NULL){
            tmp = now -> next;
            now -> next = pre;
            pre = now;
            now = tmp;
        }
        now -> next = pre;
        return now;
    }
};
